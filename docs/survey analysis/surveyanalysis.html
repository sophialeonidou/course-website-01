Age Distribution

# Age distribution
data %>%
  mutate(age_group = cut(age, breaks = seq(18, 95, by = 10), include.lowest = TRUE)) %>%
  group_by(age_group) %>%
  summarize(mean_loneliness = mean(loneliness_frequency, na.rm = TRUE)) %>%
  ggplot(aes(x = age_group, y = mean_loneliness, fill = age_group)) +
  geom_bar(stat = "identity", alpha = 0.8) +
  labs(
    title = "Mean Loneliness Frequency by Age Group",
    x = "Age Group",
    y = "Mean Loneliness Frequency",
    fill = "Age Group"
  ) +
  theme_minimal() +
  theme(axis.text.x = element_text(angle = 45, hjust = 1))

Name

data %>%
  group_by(real_name_usage) %>%
  summarize(mean_sharing_comfort = mean(sharing_comfort, na.rm = TRUE)) %>%
  mutate(percentage = mean_sharing_comfort / sum(mean_sharing_comfort) * 100) %>%
  ggplot(aes(x = "", y = percentage, fill = as.factor(real_name_usage))) +
  geom_bar(stat = "identity", width = 1) +
  coord_polar("y", start = 0) +
  labs(
    title = "Average Sharing Comfort by Real Name Usage",
    fill = "Real Name Usage Frequency",
    y = "Percentage",
    x = NULL
  ) +
  theme_void() +
  theme(legend.position = "right")
